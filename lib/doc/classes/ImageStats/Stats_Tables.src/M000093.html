      def refresh(table_key)
        if ((self.get_name_short != nil) && (table_key != nil))
         if (!@table_id)
            tables = Stat_Table_List.new(@@msh)
            @table_id = tables.get_table_id( self.get_name_short )
          end
          @value = @@msh.get_stats_data(self.get_name_short, table_key)
        end
      end
