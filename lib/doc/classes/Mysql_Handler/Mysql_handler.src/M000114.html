    def get_webcam_id( camera_name )
      query = "SELECT `id` FROM `src_camera_table` WHERE `camera_name` LIKE \"%" + camera_name + "%\" ;"
      res = @dbh.query(query)
      if (res.num_rows > 0)
        row = res.fetch_row
        return row[0]
      else
        raise "unable to find camera ID, \n query was -: #{query}" 
      end
    end
