    def get_base_url(camera)
      query = " SELECT `camera_url`
                      FROM `src_camera_table`
                      WHERE `id` =  '" + camera.to_s+"'"
      res = @dbh.query(query)
      if (res.num_rows > 0)
        row = res.fetch_row
        return row[0]
      else
        raise "unable to find camera URL, \n query was -: #{query}" 
      end
    end
